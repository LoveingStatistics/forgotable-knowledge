def get_counts(sequence):
    counts = {}
    for x in sequence:
        if x in counts:
            counts[x] += 1
        else:
            counts[x] = 1
    return counts

'''自定义排序函数'''
def top_counts(count_dict, n=10):
    value_key_pairs = [(count, tz) for tz, count in count_dict.items()] #先分解成元组，再排序
    value_key_pairs.sort(reverse=True)
    return value_key_pairs[:n]
